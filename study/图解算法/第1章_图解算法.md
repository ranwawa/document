找出有序数组中的834857元素下标

```javascript
var a = [];
var t = 955555;
for (var i = 0; i < 1234567; i++) {
  a.push(i);
}
// 普通思路
console.time('tranvers');
var i = -1;
var length = a.length;
for (let index = 0; index < length; index++) {
  if(t === a[index]) {
    i = index;
    break;
  }
}
console.log(i);
console.timeEnd('tranvers');

// 二分查找思路
console.time('binary');
function find(arr, t, start, end) {
  if(arr.length < 1) { return -1 }
  index = Math.floor(start + (end - start) / 2);
  const ele = arr[index];
  if(t === ele) {
    return index;
  } else if (t < ele) {
    return find(arr, t, start, index);
  } else if (t > ele) {
    return find(arr, t, index, end);
  }
}
var b = find(a, t, 0, a.length);
console.log(b);
console.timeEnd('binary');

```
